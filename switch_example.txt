#include <iostream>

using namespace std;

int main()
{
   int valor;
   
   cout << "Ingresa un valor del 1 al 7: ";
   cin >> valor;
   
   switch (valor){
    case 1: cout << "Lunes";
    break;
    case 2: cout << "Martes";
    break;
    case 3: cout << "Miércoles";
    break;
    case 4: cout << "Jueves";
    break;
    case 5: cout << "Viernes";
    break;
    case 6: cout << "Sábado";
    break;
    case 7: cout << "Domingo";
    break;
    default: cout << "Opción invalida";
    
    
   }
        
  return 0;
}
