// Example program
#include <iostream>

using namespace std;

int main()
{
   int valor;
   
   cout << "Ingresa un valor del 1 al 7: ";
   cin >> valor;
   
   if(valor==1)
       cout << "Lunes";   
    else if(valor==2)
        cout << "Martes";   
       else if(valor==3)
            cout << "Miércoles";
            else if(valor==4)
                cout << "Jueves";   
               else if(valor==5)
                    cout << "Viernes";   
                   else if(valor==6)
                        cout << "Sábado";
                       else if(valor==7)
                            cout << "Domingo";
                            else
                                cout << "Opción invalida";
        
  return 0;
}
